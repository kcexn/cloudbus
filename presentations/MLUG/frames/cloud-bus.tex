	\only<5>{
		\Large \textbf{Why don't we use SIP in every IT network?} \normalsize \vspace{1em}
		\begin{itemize}
			\item Protocol design suffers from telco corpospeak. \scriptsize (\emph{not cool}, nobody wants to %
			read about using a telephone to \emph{talk to people}). \normalsize
			\item Protocol design also appears to suffer a bit from \emph{Conway's law} \scriptsize (The design of a system mirrors %
			the communication structure of the organization that made it). \normalsize
			\item A minimal working example of a SIP stack has too many---pieces. \scriptsize(P-CSCF, I-CSCF, S-CSCF...) \normalsize
		\end{itemize}
	}
